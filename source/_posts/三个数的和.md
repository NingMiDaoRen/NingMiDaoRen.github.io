# 三数之和



## 题目描述

给定一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？找出所有满足条件且不重复的三元组。

注意：答案中不可以包含重复的三元组。

 

## 测试用例

示例：

给定数组 nums = [-1, 0, 1, 2, -1, -4]，

满足要求的三元组集合为：
[
  [-1, 0, 1],
  [-1, -1, 2]
]

## 解题思路

本道题的突破口就在于，**1 怎么加和**，**2 去重**

**1 加和方式**

三个数求和是两个数求和的拓展，那么如果要找a+b=0的组合，我相信大家都会。无外乎**定住a这个数字不动**，开始一步一步地找b。

我们可以假定 i 这个位置的数字不动，left【l】= i+1，right【r】 = nums.length()-1,通过移动左右指针的方式求和。

![](https://images.gitee.com/uploads/images/2020/0222/214219_53acc86d_5550632.png)

### 移动策略

首先将数组变成有序的【升序】，左右两个指针的移动取决于加和(sum)的值

```python
if sum == 0:
    "将nums[i],nums[left],nums[right]加到数组中，并将此数组追加到答案数组中"
elif sum > 0:
    #这里是要排序的原因
    #数大了，要变得相对小
    right-=1
elif sum < 0:
    left+=1;
	
```



**2 去重**

首先要弄清楚重复的原因：**输入数组中有重复的数字**，如果没有重复的数字，那么就不可能出现重复的情况。

既然这样，解决方案也很清晰，因为**数组是排好序的**，就会导致**相同数字是相邻的**，只要跳过数值相同的相邻元素就可以保证唯一性了。

都有哪些位置需要判定重复呢？

1. 基准下标 i 和 i-1【第一个数不考虑重复问题，从第二个数开始】
2. left 和 left+1
3. right 和 right - 1



## 代码

```C++
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

vector<vector<int> > threeSum(vector<int>& nums);
void print(vector<vector<int> >matrix);

int main()
{
	vector<int> test = {0,0,0,0};
	print(threeSum(test));
	
	return 0;
}

vector<vector<int> > threeSum(vector<int>& nums) 
{
	int sum = 0;
    vector<vector<int> > ans;
    vector<int> arr;
    
    if(nums.size()>=3)
	{
		sort(nums.begin(),nums.end());
	    for(int i = 0;i<nums.size();i++)
	    {
	    	if(nums[i] >0 ) break;
	    	if(i > 0 && nums[i] == nums[i-1]) continue;
			int left = i+1,right = nums.size()-1;
			while(left<right)
			{
				sum = nums[i] + nums[left] + nums[right];
				if(sum == 0)
				{
					arr.push_back(nums[i]);
					arr.push_back(nums[left]);
					arr.push_back(nums[right]);
					ans.push_back(arr);
					arr.erase(arr.begin(),arr.end());
					while(left<right && nums[left] == nums[left+1])
					{
                        //重复性检查
						left++;
					}
					while(left<right && nums[right] == nums[right-1])
					{
                        //重复性检查
						right--;
					}
					left++;
					right--;
				}
				else if(sum < 0)
				{
					left++;
				}
				else if(sum > 0)
				{
					right--;
				}
			}
		}		
	}
	return ans;

}

void print(vector<vector<int> >matrix)
{
	int i,j;
	for(i = 0;i<matrix.size();i++)
	{
		for(j = 0;j<matrix[i].size();j++)
		{
			cout << matrix[i][j] << ",";
		}
		cout << endl;
	}
}
```

## 练习 1 答案



```c++
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

vector<vector<int>> fourSum(vector<int>& nums, int target);
void print(vector<vector<int> >matrix);

int main()
{
	int target = 0;
	vector<int> nums = {0,0,0,0};
	print(fourSum(nums,target));
	
	return 0;
}

vector<vector<int>> fourSum(vector<int>& nums, int target) 
{
    vector<vector<int> > ans;
    int sum;
    int a,b,c,d;
    if(nums.size()>=4)
    {
        sort(nums.begin(),nums.end());
        for(a = 0;a<nums.size();a++)
        {
            if(a>0 && nums[a]==nums[a-1]) continue;
            for(b = a+1;b<nums.size();b++)
            {
                if(b>a+1 && nums[b] == nums[b-1]) continue;
                c = b+1;
                d = nums.size()-1;
                while(c<d)
                {
                    sum = nums[a] + nums[b] + nums[c] + nums[d];
                    if(sum == target)
                    {
                        ans.push_back({nums[a],nums[b],nums[c],nums[d]});
                        while(c<d && nums[c] == nums[c+1]) c++;
                        while(c<d && nums[d] == nums[d-1]) d--;
                        c++;
                        d--;
                    }
                    else if(sum < target) c++;
                    else if(sum > target) d--;
                }
            }
        }		
    }
    return ans;
}

void print(vector<vector<int> >matrix)
{
	int i,j;
	for(i = 0;i<matrix.size();i++)
	{
		for(j = 0;j<matrix[i].size();j++)
		{
			cout << matrix[i][j] << ",";
		}
		cout << endl;
	}
}
```

例如，给定数组 nums = [-1，2，1，-4], 和 target = 1.

## 练习 2 答案



```c++
#include<iostream>
#include<vector>
#include<algorithm>


using namespace std;
int threeSumClosest(vector<int>& nums, int target);

int main()
{
	vector<int>nums = {-1,2,1,-4};
	int target = 1;
	cout << threeSumClosest(nums,target) << endl;
	return 0;
}

int threeSumClosest(vector<int>& nums, int target) 
{
	int sum = 0,min_val = 1000000,ans = 0;
	int i,left,right;
	
	sort(nums.begin(),nums.end());
	for(i = 0;i<nums.size();i++)
	{
		if(i>0 && nums[i] == nums[i-1]) continue;
		left = i+1;
		right = nums.size()-1;
		while(left<right)
		{
			sum = nums[i] + nums[left] + nums[right];
			if(abs(sum-target) < min_val)
			{
				min_val = abs(sum-target);
				ans = sum;
			}
			else if(sum > target) right--;
			else if(sum < target) left++;
			else
			{
				return ans;
			}
		}
	}
	return ans;
}
```
